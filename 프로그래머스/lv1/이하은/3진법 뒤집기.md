class Solution {
    fun solution(n: Int): Int {
        var N = n 
        val temp = mutableListOf<Int>()
        
        // 몫이 3보다 작을 때까지 반복
        while(true){
            if(N < 3){
                temp.add(N)
                break
            }
            
            temp.add(N % 3)
            N /= 3
        }
        
        var sum = 0
        for(i in 0 until temp.size){
            if(temp[i] == 0) continue
            sum += temp[i] * pow(3, temp.size - i - 1)
        }
        
        return sum
    }
    
    fun pow(a: Int, b: Int): Int {
        // a를 b번 곱한다. 
        var result = 1
        for(i in 0 until b){
            result *= a
        }
        return result
    }
}
