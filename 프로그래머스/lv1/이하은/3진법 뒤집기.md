class Leetcode.이재훈.Solution {
    fun solution(n: Int): Int {
        var N = n 
        val temp = mutableListOf<Int>()
        
        while(N > 0){
            temp.add(N % 3)
            N /= 3
        }
        
        var sum = 0
        for(i in 0 until temp.size){
            if(temp[i] == 0) continue
            sum += temp[i] * pow(3, temp.size - i - 1)
        }
        
        return sum
    }
    
    fun pow(a: Int, b: Int): Int {
        var result = 1
        for(i in 0 until b){
            result *= a
        }
        return result
    }
}
