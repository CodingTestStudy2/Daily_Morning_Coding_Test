    fun solution(arr: IntArray): IntArray {
        val answer = sortedSetOf<Int>() // 이진 탐색 트리 기반으로 정렬된 집합 
        val N = arr.size 
        
        for(i in 0 until N){
            for(j in 0 until N){
                if(i == j) continue
                answer.add(arr[i] + arr[j]) // O(logN)
            }
        }
        
        return answer.toIntArray()
    }
}
