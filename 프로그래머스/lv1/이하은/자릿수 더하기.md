class Solution {
    fun solution(n: Int): Int {
        var sum = 0
        for(ch in n.toString()){
            sum += ch - '0'
        }
        return sum 
    }
}

class Solution {
    fun solution(n: Int): Int {
        var input = n
        var sum = 0
        
        while(input > 0){
            sum += input % 10 
            input /= 10 
        }
        
        return sum 
    }
}
