    val hash = hashMapOf<String, MutableList<String>>()
    val nums = mutableListOf<Int>()
    var n = 0
    var answer = 0
    
    fun solution(clothes: Array<Array<String>>): Int {
        for(cloth in clothes){
            val (name, type) = cloth
            hash.getOrPut(type) { mutableListOf() }.add(name)
        }
        
        n = hash.size
        arr = Array(n) { 0 }
        selected = Array(n) { false }
        
        for((key, values) in hash){
            nums.add(values.size)
        }
        
        for(k in 1..n){
            // n개 중에 k개의 종류 선택
            // 각 종류에서 하나씩 선택
            dfs(0, 0, k)
            arr = Array(n) { 0 }
            selected = Array(n) { false }
        }
        
        return answer
    }
    
    // 3 2 4 5 -> 각 종류의 리스트 크기 
    // n개 중에 k개 선택
    fun dfs(idx: Int, cnt: Int, k: Int){
        if(cnt == k){
            var temp = 1
            for(i in 0 until k){
                temp *= nums[arr[i]] // 각 종류의 리스트 크기 
            }
            answer += temp 
            return 
        }
        
        for(i in idx until n){
            if(!selected[i]){
                selected[i] = true
                arr[cnt] = i
                dfs(i + 1, cnt + 1, k) // idx가 아니라 i (오타 주의) 
                selected[i] = false
            }
        }
    }
}
