    public boolean solution(String[] phones) {
        // 크기 순으로 정렬 
        Arrays.sort(phones);
        
        // 접두어 확인 
        for(int i = 0; i < phones.length - 1; i++){
            if(phones[i + 1].startsWith(phones[i])){
                return false;
            }
        }
        
        return true;
    }
}

- M: phones[i + 1], phones[i] 중에 더 짧은 문자열의 길이 (최대 20)
