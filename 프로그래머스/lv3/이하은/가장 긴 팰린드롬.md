## substring, reversed (시간초과)

```kotlin
class Leetcode.이재훈.Solution {
    fun solution(s: String): Int {
        var answer = 0
        var n = s.length
        var left = 0
        var right = n
        
        while(n > 0){
            while(right <= s.length){
                if(isPalindrome(s.substring(left, right))){
                   return n
                }
                
                left++
                right++
            }
            
            n--
            left = 0 
            right = n 
        }
        
        return answer
    }
    
    fun isPalindrome(s: String): Boolean {
        return s == s.reversed()
    }
}
```

## 투포인터로 절반 길이까지만 확인 

```kotlin
class Leetcode.이재훈.Solution {
    fun solution(s: String): Int {
        var answer = 0
        
        // 최대 길이에서 하나씩 줄여나가면서 
        for(len in s.length downTo 1){
            // [start, start + len) 구간의 문자열 검사 
            var start = 0 
            while(start + len <= s.length){
                if(isPalindrome(s, start, start + len - 1)){
                   return len // 최대 길이 리턴 
                }
                
                // 슬라이딩 윈도우
                // 문자열 길이 고정된 상태로 start++ 
                start++
            }
        }
        
        return answer
    }
    
    fun isPalindrome(s: String, left: Int, right: Int): Boolean {
        var st = left 
        var en = right 
        
        // 절반 길이까지만 검사 
        while(st <= en){
            if(s[st++] != s[en--]) return false
        }
        
        return true 
    }
}
```
