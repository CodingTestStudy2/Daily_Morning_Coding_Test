## 시간 초과 - DFS

```kotlin 
/*
(x, y) ~ (r, c) 
-> 이동 거리 k가 되는 모든 경우의 수 
-> 사전 순으로 가장 앞서는 경로 리턴 
-> 불가능한 경우는 impossible 리턴  
-> S, E 포함해서 같은 격자 두번 이상 방문 OK 

d l r u -> 사전순으로 탐색
*/
class Solution {
    var N = 0
    var M = 0
    var R = 0
    var C = 0
    var K = 0
    val dx = listOf(1, 0, 0, -1)
    val dy = listOf(0, -1, 1, 0)
    val dir = listOf('d', 'l', 'r', 'u')
    var found = false
    var answer = ""
    
    fun solution(n: Int, m: Int, x: Int, y: Int, r: Int, c: Int, k: Int): String {
        N = n
        M = m
        R = r
        C = c
        K = k

        dfs(x - 1, y - 1, 0, "")
        
        return if(answer.isEmpty()) "impossible" else answer
    }
    
    fun dfs(x: Int, y: Int, dist: Int, result: String){
        if(x == R - 1 && y == C - 1){
            if(dist == K && !found){
                answer = result
                found = true
                return
            }
        }
        
        if(dist > K) return // 이렇게만 하면 안 된다! 
        
        for(i in 0..3){
            val nx = x + dx[i]
            val ny = y + dy[i]
            if(nx < 0 || nx >= N || ny < 0 || ny >= M) continue 
            
            dfs(nx, ny, dist + 1, result + dir[i])
            
            if(found) return 
        }
    }
}
```

## 재귀 함수의 탈출 조건 구체화 

```kotlin 
/*
(x, y) ~ (r, c) 
-> 이동 거리 k가 되는 모든 경우의 수 
-> 사전 순으로 가장 앞서는 경로 리턴 
-> 불가능한 경우는 impossible 리턴  
-> S, E 포함해서 같은 격자 두번 이상 방문 OK 

d l r u -> 사전순으로 탐색
*/
import kotlin.math.*

class Solution {
    var N = 0
    var M = 0
    var R = 0
    var C = 0
    var K = 0
    val dx = listOf(1, 0, 0, -1)
    val dy = listOf(0, -1, 1, 0)
    val dir = listOf('d', 'l', 'r', 'u')
    var found = false
    var answer = ""
    
    fun solution(n: Int, m: Int, x: Int, y: Int, r: Int, c: Int, k: Int): String {
        N = n
        M = m
        R = r
        C = c
        K = k
        
        dfs(x, y, 0, "")
        
        return if(found) answer else "impossible"  
    }
    
    fun dfs(x: Int, y: Int, dist: Int, result: String) {
        val remainDist = abs(x - R) + abs(y - C)
        
        // 도착점까지 거리가 멀어서 불가능 
        if(K - dist < remainDist) return
        
        // 도착점까지 남은 거리가 홀수라면 불가능 
        // K - dist >= remainDist -> 왔다갔다 해서 도착점에 도달 (거리는 짝수)
        if((K - dist - remainDist) % 2 == 1) return
        
        if(dist == K && x == R && y == C){
            found = true 
            answer = result 
            return 
        }
        
        for(i in 0..3){
            val nx = x + dx[i]
            val ny = y + dy[i]
            if(nx < 1 || nx > N || ny < 1 || ny > M) continue 
            
            dfs(nx, ny, dist + 1, result + dir[i])
            
            if(found) return
        }
    }
}
```
