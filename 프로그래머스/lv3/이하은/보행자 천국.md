## bottom-up (반복문)

```java
/*
행 m, 열 n 
오른쪽 또는 아래로만 이동 가능 
1 -> 통행 금지 
2 -> 이동 방향 회전 금지 
이동 가능한 전체 경로 수 출력 
*/
class Solution {
    int answer = 0;
    int MOD = 20170805;
    
    public int solution(int m, int n, int[][] cityMap) {
        // (i, j)로 이동 가능한 경로의 수 저장 (방향 포함)
        int[][][] dp = new int[m + 1][n + 1][2];
        
        // 0: 아래쪽으로 이동 
        // 1: 오른쪽으로 이동 
        dp[1][1][0] = 1;
        dp[1][1][1] = 1;
        
        for(int i = 1; i <= m; i++){
            for(int j = 1; j <= n; j++){
                if(cityMap[i - 1][j - 1] == 0){
                    // 위에서 아래로 or 왼쪽에서 오른쪽으로 
                    dp[i][j][0] += (dp[i-1][j][0] + dp[i][j-1][1]) % MOD; 
                    dp[i][j][1] += (dp[i-1][j][0] + dp[i][j-1][1]) % MOD; 
                }else if(cityMap[i - 1][j - 1] == 1){
                    dp[i][j][0] = 0;
                    dp[i][j][1] = 0;
                }else{
                    // 위에서 아래로 이동
                    dp[i][j][0] += dp[i-1][j][0];
                    
                    // 왼쪽에서 오른쪽으로 이동 
                    dp[i][j][1] += dp[i][j-1][1]; 
                }
            }
        }
        
        return dp[m][n][0];
    }
}
```

## top-down (dfs 재귀 호출)

```java
class Solution {
    int MOD = 20170805;
    int MAX = 501; 
    int[][][] dp = new int[MAX][MAX][2];
    int[] dx = {0, 1};
    int[] dy = {1, 0};
    
    int[][] arr; 
    int M = 0; 
    int N = 0; 
    
    public int solution(int m, int n, int[][] cityMap) {
        arr = cityMap;
        M = m; 
        N = n; 
        
        initMemo();
        return dfs(0, 0, 0);
    }
    
    int dfs(int x, int y, int dir){
        if(x == M - 1 && y == N - 1) return 1;

        // dp 테이블에 저장된 값이 있으면 그대로 리턴 
        if(dp[x][y][dir] != -1) return dp[x][y][dir];
        
        // 현재 가지에서 가능한 이동 경로의 수 
        int sum = 0;

        for(int i = 0; i < 2; i++){
            int nx = x + dx[i];
            int ny = y + dy[i];
            if(nx < 0 || nx >= M || ny < 0 || ny >= N) continue;
            
            // 이전 칸이 2일 때는, 방향이 같은 경우에만 이동 
            if(arr[x][y] == 0 || (arr[x][y] == 2 && dir == i)){
                sum += dfs(nx, ny, i) % MOD; 
                sum %= MOD; 
            }
        }
        
        return dp[x][y][dir] = sum; // 메모이제이션 
    }
    
    void initMemo(){
        for(int i = 0; i < M; i++){
            for(int j = 0; j < N; j++){
                for(int d = 0; d < 2; d++){
                    dp[i][j][d] = -1;
                }
            }
        }
    }
}
```
